\begin{theorem}
    \label{thm:valid_upper_bound}
    For any $\discretization, \itinSetSuper$, if the model $\ubm(\discretizationFull, \model{\discretization, \itinSetSuper})$ has a non-empty feasible region, the optimal objective value is an upper bound for the original model $\model{\discretizationFull}$.
\end{theorem}

% Page 1: 结论
\begin{frame}{结论}
\begin{block}{研究问题}
考虑顾客选择的服务网络设计——供给决策与需求分配的联合优化。
\end{block}

\vspace{0.2cm}
\begin{block}{方法贡献：IA-DR 框架}
\begin{itemize}
    \item \hl{下界模型}：通过降维快速得到紧的下界
    \item \hl{上界模型}：基于下界提供可行解
    \item \hl{动态细化}：检测并消除松弛产生的假象，保证有限步收敛到全局最优
\end{itemize}
\end{block}

\vspace{0.2cm}
\begin{block}{实验验证}
\begin{itemize}
    \item 在大规模实例上，IA-DR 显著优于直接求解原始模型。
\end{itemize}
\end{block}
\end{frame}

\section{方法框架}

\begin{frame}{IA-DR 框架}

% \centering
% \large 不是一次性求完整大模型，而是 \hl{先做紧下界，再做上界修复，再做定向细化}。
% \end{block}

\vspace{0.15cm}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.65\textwidth}
\begin{block}{三个模块}
\small
\begin{itemize}
    \item \hl{下界}：压缩问题维度，得到松弛解
    \item \hl{上界}：从松弛解恢复可行解
    \item \hl{细化}：在必要位置加时间点，使下界更靠近原问题
\end{itemize}
\end{block}

\begin{frame}{下界-空间维度聚合}
\small

\begin{columns}[T,totalwidth=\textwidth]
\column{0.55\textwidth}
\small
\begin{block}{更小的时空网络}
\begin{itemize}
    \item 只保留\hl{部分时间点}，代表原始时间线上的一个\hl{区间}
    \item 将原始的\hl{服务}弧映射到新的时空网络上，得到\hl{聚合服务}弧
\end{itemize}
\end{block}

\begin{block}{性质}
\begin{itemize}
    \item 映射后服务的开始、结束时间更加\hl{乐观}
    \item 变量数量减少的同时保证可行域\hl{严格包含}原问题的可行域
\end{itemize}
\end{block}

\begin{frame}{下界-行程维度聚合}
\small
\begin{columns}[T,totalwidth=\textwidth]
\column{0.55\textwidth}
\small
\begin{block}{更小的行程空间}
\begin{itemize}
    \item 使用\hl{聚合行程}代替相同\hl{路径、时间段}的行程
    \item \hl{票价}取组内最大值
    \item \hl{吸引力}取物理可共存组合的乐观上界
\end{itemize}
\end{block}

\vspace{0.12cm}
\begin{block}{性质}
\begin{itemize}
    \item 降低\hl{顾客可选行程}变量数量
    \item 大幅剪除\hl{不可行}行程组合，得到\hl{更紧的下界}
    \item 仍保证可行域\hl{严格包含}原问题的可行域
\end{itemize}

\end{block}

\begin{frame}{上界修复}
\begin{block}{核心思想}
\centering
\large 在\hl{完整网络}上，锁定下界模型的资源分配，找到局部最优。
\end{block}

\vspace{0.3cm}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.65\textwidth}
\begin{block}{做法}
\begin{itemize}
    \item 从 LBM 解中提取每条航段、每种机型的服务频率
    \item 锁定频率，在完整时空网络上求解
    \item 问题简化为：给定每个服务的资源量，找最优时刻表
\end{itemize}
\end{block}
\end{columns}
\end{frame}


\begin{frame}{动态细化网络}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.52\textwidth}
\begin{block}{检测$\to$加点}
\begin{enumerate}
    \item 检测下界解中\hl{不合法}的部分(artifacts)
    \item 对\hl{出问题的位置}加时间点
\end{enumerate}
\end{block}
\end{columns}

\vspace{0.18cm}
\begin{alertblock}{理论保证}
    $$\text{下界解不合法}\iff\text{可以通过本算法加点}$$
    $\to$ 算法会在有限步内\hl{\textbf{收敛到全局最优解}}
\end{alertblock}
\end{frame}

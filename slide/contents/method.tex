\begin{frame}{核心概念——超级行程}

\begin{block}{构造方式}
\begin{itemize}

    \item 票价：取组内乐观值（最大值）

\end{itemize}

\vspace{0.2cm}

\begin{itemize}
    \item 大幅压缩行程空间维度
    \item 保持松弛的有效性
    \item 限制不兼容行程的收入累加
\end{itemize}
\end{block}

% Page 8: 超级行程的数学表示
\begin{frame}{超级行程的数学表示}
\begin{block}{路径签名等价关系}
\vspace{-0.2cm}
\[r\sim r' \;\Longleftrightarrow\; \pi(r)=\pi(r')\]
\small 相同聚合路径签名的原始行程被归入同一超级行程 $R$。
\end{block}

\vspace{0.2cm}
\begin{block}{超级行程吸引力}
\vspace{-0.2cm}
\[\beta_{Rp\ell}=\max_{\mathcal{I}\in\mathcal{C}(R)}\sum_{\mathcal{O}\in\mathcal{I}}\sum_{r\in\mathcal{O}}\alpha_{rp\ell}\]
\small
\begin{itemize}
    \item $\mathcal{C}(R)$：超级行程 $R$ 内允许的候选组合族
    \item $\beta_{Rp\ell}$ 不是组内全部行程吸引力的简单累加
    \item 它是\hl{可兼容原始行程组合}吸引力总和的乐观上界
\end{itemize}
\end{block}

\vspace{0.15cm}
\centering
\small 超级行程不是画图概念，而是有严格数学定义的聚合变量。

% Page 9: 不可实现收入与行程聚合
\begin{frame}{不可实现收入与行程聚合}
\begin{block}{问题：简单时间聚合导致不可实现收入}
\begin{itemize}
    \item LP 松弛甚至整数解可能累加来自\hl{不兼容行程}的收入
    \item 同一时段内多条行程共享同一服务弧 → 收入被重复计算
    \item 导致下界过于乐观，间隙难以收紧
\end{itemize}
\end{block}

\vspace{0.3cm}
\begin{block}{解决：超级行程构造限制吸引力累加}
\begin{itemize}
    \item 超级行程的吸引力基于\hl{可兼容组合的乐观上界}构造
    \item 不可同时实现的行程不会被无约束地全部累加
    \item 结果：得到\hl{更紧的下界}，加速算法收敛
\end{itemize}
\end{block}

% Page 10: 双侧界模型公式
\begin{frame}{双侧界模型：下界与上界}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.48\textwidth}
\begin{block}{聚合下界模型}
\small
\vspace{-0.3cm}
\[\min\sum_{a,f} C_{af}x_{af}-\sum_{m,p,R,\ell}D_{mp}\mathit{TP}_{R\ell}w_{Rp\ell}\]
\vspace{-0.3cm}

关键替换：
\vspace{-0.2cm}
\[\alpha^0_{mp}w_{Rp\ell}\le \beta_{Rp\ell}w^0_{mp}\]
\vspace{-0.3cm}

\small 在聚合网络和超级行程空间上求解，得到原问题\hl{有效下界}。
\end{block}

\column{0.48\textwidth}
\begin{block}{完整网络上界修复}
\small
频率固定：
\vspace{-0.2cm}
\[\sum_{a\in \hat{\mathcal{A}}(s)}x_{af}=n^*_{sf}\]
\vspace{-0.3cm}

超级行程收入约束：
\vspace{-0.2cm}
\[\sum_{r\in S(R)}D_{mp}\mathit{TP}_{r\ell}w_{rp\ell}\le D_{mp}\mathit{TP}_{R\ell}\mathit{sw}_{Rp\ell}\]
\vspace{-0.3cm}

\small 在完整时空网络上求解，生成原问题可行解，给出\hl{有效上界}。
\end{block}
\end{columns}

\vspace{0.2cm}
\centering
\small 下界与上界之差 = 当前最优性间隙，指导细化方向

% Page 11: 可实现性检验与时间点添加
\begin{frame}{可实现性检验与时间点添加}
\begin{block}{松弛解在完整网络中的三类非法情形}
\begin{enumerate}
    \item<1-> \hl{不可能出发}：聚合网络中资源过早可用，完整网络中无法实现
    \item<2-> \hl{间隔出发违规}：同一物理路段的出发间隔不满足最小间隔约束
    \item<3-> \hl{不可实现收入}：超级行程声称的收入无法由原始行程组合实现
\end{enumerate}
\end{block}

\vspace{0.2cm}
\begin{block}{定向时间点添加}
\begin{itemize}
    \item 针对检测到的非法情形，添加对应时间点
    \item 更新路径签名与超级行程划分，排除当前非法松弛解
\end{itemize}
\end{block}

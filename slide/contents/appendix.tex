% ===== 附录 A：超级行程吸引力上界公式 =====
\begin{frame}{附录 A：超级行程吸引力上界}
\small 超级行程不是画图概念，而是有严格数学定义的聚合变量。

\vspace{0.1cm}
\begin{block}{路径签名等价关系}
\vspace{-0.2cm}
\[r \eqRelByPathSig r' \;\Longleftrightarrow\; \pathSig{r}=\pathSig{r'}\]
\small 相同聚合路径签名的原始行程被归入同一超级行程 $R$。
\end{block}

\vspace{0.15cm}
\begin{block}{吸引力上界}
\vspace{-0.2cm}
\[\attrSuper_{Rp\ell}=\max_{\itinCombSet\in\compatSet{R}}\sum_{\eqClassByOrigPath\in\itinCombSet}\sum_{r\in\eqClassByOrigPath}\attr_{rp\ell}\]
\small
\begin{itemize}
    \item $\compatSet{R}$：覆盖可兼容原始行程组合的候选组合族
    \item $\attrSuper_{Rp\ell}$：可兼容组合吸引力总和的乐观上界
\end{itemize}
\end{block}

\vspace{0.1cm}
\centering
\small 该构造避免同一超级行程内所有原始行程被无约束累加，从而\hl{抑制不可实现收入}。
\end{frame}


% ===== 附录 A（续）：Attraction 聚合详细公式 =====
\begin{frame}{附录 A（续）：Attraction 聚合详细公式}
\tiny
\begin{block}{路径签名分组}
原始行程 $r$ 的聚合路径签名：$\pathSig{r}=(o_1,d_1,\bar{t}_1),\ldots,(o_k,d_k,\bar{t}_k)$。
按 $\pathSig{r}$ 分组得到等价类 $\eqClassByPathSig$，每个等价类构造一个超级行程 $R$。
\end{block}
\vspace{0.05cm}
\begin{block}{组内再按原始路径签名分组}
$\eqClassByPathSig_{(R)}$ 内按原始路径 $\origPath{r}$ 再分组得 $\eqClassByOrigPath^R$。
同一 $\eqClassByOrigPath^R_r$ 内的行程使用完全相同的物理路径，可同时运营。
\end{block}
\vspace{0.05cm}
\begin{block}{Attraction 计算}
\vspace{-0.15cm}
\begin{equation*}
\attrSuper_{Rp\ell} = \max_{\itinCombSet \subseteq \eqClassByPathSig_{(R)}/\eqRelByOrigPath,\;|\itinCombSet|\le\combLimit_R}
\sum_{\mathscr{O}\in\itinCombSet}\sum_{r\in\mathscr{O}}\attr_{rp\ell}
\qquad
\combLimit_R = \prod_{(o,d,\bar{t})\in\pathSig{R}} \min\{\departLimit_{od\bar{t}},\;\freq_{od}\}
\end{equation*}
\end{block}
\vspace{0.05cm}
\begin{alertblock}{}
\centering\scriptsize
$\combLimit_R$ 是合法共存服务数的上界 $\Rightarrow$ $\attrSuper_{Rp\ell} \ge$ 任何可行整数解的吸引力 $\Rightarrow$ \textbf{下界有效}
\end{alertblock}
\end{frame}


% ===== 附录 B：聚合下界模型 formulation =====
\begin{frame}{附录 B：聚合下界模型}
\small
\begin{block}{目标函数}
\vspace{-0.3cm}
\[\min\sum_{a\in\arcSet^F}\sum_{f\in\fleetSet} C_{af}x_{af}-\sum_{m,p,R,\ell}D_{mp}\mathit{TP}_{R\ell}w_{Rp\ell}\]
\end{block}
\vspace{0.1cm}
\begin{block}{关键约束}
\begin{itemize}
    \item 资源流守恒、资源可用性、最小间隔（聚合网络版本）
    \item 超级行程选择约束：$\alpha^0_{mp}w_{Rp\ell}\le \beta_{Rp\ell}w^0_{mp}$
    \item 聚合容量约束
\end{itemize}
\end{block}
\vspace{0.1cm}
\begin{block}{性质}
\begin{itemize}
    \item 模型 $\model{\discretization,\itinSetSuper}$ 是原始模型的有效松弛
    \item 最优值 $\le$ 原始模型最优值（下界）
\end{itemize}
\end{block}
\end{frame}


% ===== 附录 C：上界修复模型的强化约束 =====
\begin{frame}{附录 C：上界修复模型的强化约束}
\scriptsize
\begin{block}{核心思想}
在完整网络 $\full{\net}$ 上求解，锁定 LBM 的资源分配频率，注入超级行程收入约束加速收敛。
\end{block}
\vspace{0.1cm}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.48\textwidth}
\textbf{1. 频率固定}
\vspace{-0.1cm}
\begin{equation*}
\sum_{a\in\full{\segArcSet{s}}}\fleetAssignVar_{af} = \freq_{s,f}^{*} \quad \forall s,f
\end{equation*}
锁定每条物理路段每种资源类型的服务次数，将问题简化为排班修复。

\column{0.48\textwidth}
\textbf{2. 超级行程选择约束}

引入超级行程份额变量 $\shareSVar_{Rp\ell}$：
\vspace{-0.1cm}
\begin{align*}
& \outAttr_{mp}\shareSVar_{Rp\ell} \le \attrSuper_{Rp\ell}\outsideShareVar_{mp} \\
& \tfrac{\adjOutAttr_{mp}}{\outAttr_{mp}}\outsideShareVar_{mp} + \sum_{R,\ell}\tfrac{\adjAttrSuper_{Rp\ell}}{\attrSuper_{Rp\ell}}\shareSVar_{Rp\ell} \le 1 \\
& \sum_{m,p,\ell}\sum_{R\in\itinSuperUsingArc{m}{a}}\demand_{mp}\shareSVar_{Rp\ell} \le \sum_{a'\in\Phi_{\arcSet}^{-1}(a)}\sum_f\fleetCap_f\fleetAssignVar_{a'f}
\end{align*}

\vspace{0.1cm}
\textbf{3. 超级行程收入约束}
\vspace{-0.1cm}
\begin{equation*}
\sum_{r\in \eqClassByPathSig_{(R)}}\demand_{mp}\fare_{r\ell}\shareVar_{rp\ell} \le \demand_{mp}\fare_{R\ell}\shareSVar_{Rp\ell}
\end{equation*}
底层原始行程总收入不超过超级行程层面的收入上界。
\end{columns}
\end{frame}


% ===== 附录 D：三类非法情形详细 =====
\begin{frame}{附录 D：松弛解可实现性检验——三类非法情形}
\scriptsize
\begin{columns}[T,totalwidth=\textwidth]
\column{0.32\textwidth}
\begin{block}{\centering 1. 不可能出发\\（时序违规）}
\begin{itemize}\setlength\itemsep{0.05cm}
    \item 聚合后某弧的到达时间 $\le$ 出发时间（逆弧效应）
    \item 资源被提前到达，虚假满足出发需求
    \item 修复：插入真实到达时间 $t_{\mathrm{dep}}+\travelTime$ 到 $\discretization_i$
\end{itemize}
\end{block}

\column{0.32\textwidth}
\begin{block}{\centering 2. 间隔出发违规\\（间隔违规）}
\begin{itemize}\setlength\itemsep{0.05cm}
    \item 多个物理服务被聚合到同一时间窗
    \item 实际出发间隔 $<$ 最小间隔 $\minInterdepTime_{ij}$
    \item 修复：插入冲突服务的真实出发时间到 $\discretization_i$
\end{itemize}
\end{block}

\column{0.32\textwidth}
\begin{block}{\centering 3. 不可实现收入\\（聚合收入无法物理实现）}
\begin{itemize}\setlength\itemsep{0.05cm}
    \item 超级行程的吸引力无法被任何原始行程组合实现
    \item 离散化太粗，无法区分冲突的原始行程
    \item 修复：插入相关行程的物理时间点，打破等价类
\end{itemize}
\end{block}
\end{columns}

\vspace{0.2cm}
\begin{alertblock}{}
\centering\small
定理：松弛解非法 $\Leftrightarrow$ 存在至少一类非法情形。持续消除 $\Rightarrow$ 有限步收敛。
\end{alertblock}
\end{frame}

\section{问题背景}

% Page 1: 服务网络设计
\begin{frame}{服务网络设计}
\begin{block}{定义}
\justifying
服务网络设计是指在给定物理网络上，决定\hl{开设哪些服务}（路线、频率、时刻、资源分配），以最优化运营目标的规划问题。
\end{block}

\vspace{0.2cm}
\begin{block}{应用场景}
\begin{itemize}
    \item \hl{航空}：航班时刻表与机队分配
    \item \hl{铁路}：列车开行方案与编组计划
    \item \hl{城际客运}：线路频率与车辆调度
\end{itemize}
\end{block}

\vspace{0.2cm}
\begin{block}{核心目标}
\centering
在满足运营约束下：$\min_{\text{服务方案}} \{\text{运营成本}-\text{服务收益}\}$
\end{block}
\end{frame}


% Page 2: 为什么要考虑顾客选择
\begin{frame}{为什么要考虑顾客选择}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.45\textwidth}
传统服务网络设计将需求视为\hl{外生固定输入}。

\vspace{0.2cm}
\begin{block}{客运场景的特殊性}
\justifying
\begin{itemize}
    \item 服务方案决定了可选行程
    \item 顾客选择决定了实际收益
    \item \hl{供给}与\hl{需求}存在反馈耦合
\end{itemize}
\end{block}

% Page 3: 基准模型
\begin{frame}{基准模型：运营决策与顾客选择的耦合}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.55\textwidth}
\begin{block}{目标函数}
\small
\vspace{-0.3cm}
\[\min \sum_{a\in \hat{\mathcal{A}}^F}\sum_{f\in\mathcal{F}} C_{af}x_{af}-\sum_{m,p,r,\ell}D_{mp}\mathit{TP}_{r\ell}w_{rp\ell}\]
\end{block}
\vspace{-0.1cm}
\begin{block}{关键耦合约束}
\small
选择约束：
\vspace{-0.2cm}
\[\alpha^0_{mp} w_{rp\ell} \le \alpha_{rp\ell} w^0_{mp}\]
\vspace{-0.4cm}

容量约束：
\vspace{-0.2cm}
\[\sum_{m,p,\ell}\sum_{r\in R_m(a)} D_{mp}w_{rp\ell}\le \sum_f Q_f x_{af}\]
\end{block}

\column{0.42\textwidth}
\begin{block}{变量含义}
\small
\begin{itemize}
    \item $x_{af}\in\{0,1\}$：服务弧 $a$ 是否由资源 $f$ 运营
    \item $w_{rp\ell}\ge 0$：顾客选择行程 $r$ 的需求份额
    \item 目标 = 运营成本 $-$ 票务收入
\end{itemize}
\end{block}

\vspace{0.2cm}
\begin{alertblock}{\small 耦合是难点}
\small
选择约束刻画顾客如何分配需求；容量约束把 $w$ 与 $x$ 连接起来。这种耦合是模型难求解的根源。
\end{alertblock}
\end{columns}
\end{frame}


% Page 4: 完整模型
\begin{frame}{完整模型 SN-CC$(\hat{\discretization})$}
\scriptsize
\textbf{目标函数}（最小化 运营成本 $-$ 票务收入）
\vspace{-0.2cm}
\tikz[remember picture]\coordinate(obj-start);%
\begin{equation*}
\min \sum_{a\in \full{\flightArc}}\sum_{f\in\fleetSet} \cost_{af}\fleetAssignVar_{af}
- \sum_{m\in\marketSet}\sum_{p\in\psgTypeSet}\sum_{r\in\itinInMarket_m}\sum_{\ell\in\fareClassSet} \demand_{mp}\fare_{r\ell}\shareVar_{rp\ell}
\end{equation*}
\tikz[remember picture]\coordinate(obj-end);%
\vspace{-0.4cm}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.48\textwidth}
\tikz[remember picture]\coordinate(res-start);%
\textbf{资源约束}
\vspace{-0.15cm}
\begin{align*}
& \sum_{a\in \outArcSet{i,t}} \fleetAssignVar_{af} - \sum_{a\in \inArcSet{i,t}} \fleetAssignVar_{af} = 0 & \forall (i,t),f \\
& \sum_{a\in \holdingArcTc{\checkTime}} \fleetAssignVar_{af} + \sum_{a\in \flightArcTc{\checkTime}} \fleetAssignVar_{af} \le \fleetAvail_f & \forall f \\
& \sum_{f}\sum_{a\in \arcDuringSepTime_{ijt}} \fleetAssignVar_{af} \le 1 & \forall (i,j),t \\
& \sum_{f}\sum_{a\in \segArcSet{s}} \fleetAssignVar_{af} = \freq_s & \forall s
\end{align*}
\tikz[remember picture]\coordinate(res-end);%

\column{0.48\textwidth}
\tikz[remember picture]\coordinate(choice-start);%
\textbf{顾客选择约束}
\vspace{-0.15cm}
\begin{align*}
& \outAttr_{mp}\shareVar_{rp\ell} \le \attr_{rp\ell}\outsideShareVar_{mp} & \forall m,p,r,\ell \\
& \tfrac{\adjOutAttr_{mp}}{\outAttr_{mp}} \outsideShareVar_{mp} + \sum_{r,\ell} \tfrac{\adjAttr_{rp\ell}}{\attr_{rp\ell}}\shareVar_{rp\ell} = 1 & \forall m,p \\
& \sum_{m,p,\ell}\sum_{r\in \itinUsingArc{m}{a}} \demand_{mp}\shareVar_{rp\ell} \le \sum_f\fleetCap_f\fleetAssignVar_{af} & \forall a \\
& \sum_f \fleetAssignVar_{af} \ge \shareVar_{rp\ell} & \forall a,r,p,\ell
\end{align*}
\tikz[remember picture]\coordinate(choice-end);%
\end{columns}
\vspace{0.1cm}
\tikz[remember picture]\coordinate(coupling-start);%
\textbf{变量域：} $\fleetAssignVar_{af}\in\{0,1\}$（服务弧），$\fleetAssignVar_{af}\ge0$（持有弧），$\shareVar_{rp\ell}\ge0$，$\outsideShareVar_{mp}\ge0$
\tikz[remember picture]\coordinate(coupling-end);%

% Page 5: 计算挑战
\begin{frame}{计算挑战}
\begin{block}{行程数量爆炸}
\begin{itemize}
    \item 候选行程数量随网络规模\hl{组合爆炸}增长
    \item 60个机场、259条航段 → 可产生超过\hl{百万}条候选行程
\end{itemize}
\end{block}

\vspace{0.15cm}
\begin{block}{模型结构困难}
\begin{itemize}
    \item 二元服务变量 $x$ 与连续选择变量 $w$ 强耦合
    \item 直接建立 MIP 的 LP 松弛极弱，商业求解器难以有效求解
\end{itemize}
\end{block}

\vspace{0.15cm}
\begin{block}{现有方法的局限}
\begin{itemize}
    \item DDD 框架能压缩物理时空网络，但\hl{无法处理行程维度的数量爆炸}
\end{itemize}
\end{block}
\end{frame}
